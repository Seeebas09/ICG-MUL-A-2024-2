import matplotlib.pyplot as plt import svgwrite # Lista de puntos (x, y) que forman los vértices del polígono puntos = [(1, 1), (3, 1), (4, 3), (2, 4), (1, 3)] # Separar las coordenadas x e y para uso en matplotlib x, y = zip(*puntos) # Cerrar el polígono conectando el último punto con el primero x = list(x) + [x[0]] y = list(y) + [y[0]] # Función para dibujar el polígono rasterizado def dibujar_rasterizado(): plt.figure() plt.plot(x, y, 'b-', marker='o') plt.fill(x, y, 'skyblue', alpha=0.5) plt.title('Polígono Rasterizado') plt.gca().set_aspect('equal', adjustable='box') plt.show() # Función para guardar el polígono en formato vectorial SVG def dibujar_vectorizado(): dwg = svgwrite.Drawing('poligono.svg', profile='tiny') poligono = dwg.add(dwg.polygon(points=puntos, fill='lightblue', stroke='blue', stroke_width=2)) dwg.save() # Llamar las funciones dibujar_rasterizado() # Para mostrar el polígono en formato rasterizado dibujar_vectorizado() # Para guardar el polígono en formato SVG (vectorial)
